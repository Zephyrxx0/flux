    ```typescript
    const fetchWeather = useCallback(async () => {
      const res = await fetch("/api/weather");
      if (!res.ok) throw new Error(`Weather poll failed: ${res.status}`);
      const json = await res.json();
      const weatherData = json as WeatherData;
      setWeather(weatherData);
      setLastFetchTime(Date.now());
      return weatherData;
    }, [setWeather, setLastFetchTime]);
    ```

    Add the `onImpactChange` callback (re-sim on weather condition change per D-17-04):
    ```typescript
    const onImpactChange = useCallback((impact: WeatherImpact) => {
      const adjusted = applyWeatherAdjustment(presets.normal, impact);
      initializeSim(adjusted);
    }, [initializeSim]);
    ```

    Wire the hook into the page:
    ```typescript
    const { error: weatherError, isRetrying } = useWeather(fetchWeather, { onImpactChange });
    ```
