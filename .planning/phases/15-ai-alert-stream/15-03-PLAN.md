    After:
    ```tsx
    "use client";
    import { useEffect, useCallback } from "react";
    import { useMatchPoller } from "@/hooks/useMatchPoller";
    import { useAlertStream } from "@/hooks/useAlertStream";
    import { AlertFeed } from "@/components/dashboard/AlertFeed";
    import { useLiveStore } from "@/stores/liveStore";
    import { presets } from "@/simulation/presets";

    export default function DashboardPage() {
      const initializeSim = useLiveStore((s) => s.initializeSim);
      const initialized = useLiveStore((s) => s.initialized);
      const setMatch = useLiveStore((s) => s.setMatch);

      const fetchMatch = useCallback(async () => {
        const res = await fetch("/api/match");
        if (!res.ok) throw new Error(`Poll failed: ${res.status}`);
        const json = await res.json();
        setMatch(json.match);
        return json.match;
      }, [setMatch]);

      useMatchPoller(fetchMatch);
      useAlertStream();  // <-- NEW: mount alert SSE hook per D-15-11

      useEffect(() => {
        if (!initialized) {
          initializeSim(presets.normal);
        }
      }, [initialized, initializeSim]);

      return (
        <main className="space-y-4">
          {/* Alert feed fills the main content area below MatchBanner per D-15-14 */}
          <AlertFeed />
        </main>
      );
    }
    ```

    The `useAlertStream()` hook manages its own lifecycle (connect/reconnect/cleanup) and dispatches alerts directly to the Zustand store. The AlertFeed component reads from the store reactively. Both are mounted at the dashboard page level per D-15-11 (parallel to useMatchPoller).

    **Important:** The layout.tsx already renders MatchBanner in the header and the dashboard page as children in a `<main>` element. The AlertFeed fills the main content area below MatchBanner naturally (layout.tsx handles the header position). Do NOT add MatchBanner to the dashboard page — it's already in the layout header.
