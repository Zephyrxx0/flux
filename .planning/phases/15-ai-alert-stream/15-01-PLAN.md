    **Match context from query params:**

    **`GET(request: NextRequest)` handler:**

    ```typescript
    export async function GET(request: NextRequest) {
      const encoder = new TextEncoder();
      const matchState = extractMatchState(request.nextUrl.searchParams);
      const prompt = buildAlertPrompt(zoneData, matchState);

      const stream = new ReadableStream({
        async start(controller) {
          try {
            let accumulatedJson = "";
            // Pipe Gemini stream tokens as SSE "token" events
            for await (const token of streamGeminiResponse(prompt, { signal: request.signal })) {
              accumulatedJson += token;
              controller.enqueue(
                encoder.encode(`data: ${JSON.stringify({ type: "token", text: token })}\n\n`)
              );
            }
            // Validate complete JSON тЖТ dispatch "alert" events
            const alerts = validateAlertOutput(accumulatedJson);
            for (const alert of alerts) {
              controller.enqueue(
                encoder.encode(`data: ${JSON.stringify({ type: "alert", alert })}\n\n`)
              );
            }
            // Signal cycle complete
            controller.enqueue(encoder.encode(`data: ${JSON.stringify({ type: "complete" })}\n\n`));
          } catch (error) {
            const message = error instanceof Error ? error.message : "Unknown error";
            console.error("[alert] Error:", message);
            // Check for missing API key specifically
            if (error instanceof GeminiFetchError && error.message.includes("Missing GEMINI_API_KEY")) {
              controller.enqueue(encoder.encode(
                `data: ${JSON.stringify({ type: "error", message: "API key not configured" })}\n\n`
              ));
            } else if (error instanceof GeminiRateLimitError) {
              controller.enqueue(encoder.encode(
                `data: ${JSON.stringify({ type: "error", message: "Rate limit reached тАФ will retry next cycle" })}\n\n`
              ));
            } else {
              controller.enqueue(encoder.encode(
                `data: ${JSON.stringify({ type: "error", message })}\n\n`
              ));
            }
          } finally {
            controller.close();
          }
        },
      });

      // Handle client disconnect
      request.signal.addEventListener("abort", () => {
        // signal propagates to fetch via AbortSignal; stream start() will throw on enqueue,
        // finally block closes. No extra cleanup needed per RESEARCH.md Pattern 2.
      });

      return new Response(stream, {
        headers: {
          "Content-Type": "text/event-stream; charset=utf-8",
          "Cache-Control": "no-cache, no-transform",
          Connection: "keep-alive",
          "X-Accel-Buffering": "no",     // CRITICAL: disables proxy buffering (Pitfall 4)
        },
      });
    }
    ```

    **CRITICAL RULES (from RESEARCH.md anti-patterns):**
    1. The `for await` loop MUST be inside `start(controller)` тАФ NOT before returning the Response. If the entire Gemini stream completes before `return new Response(...)`, Next.js buffers everything. (Pitfall 2)
    2. Each SSE event MUST end with `\n\n` (double newline) тАФ single `\n` silently breaks EventSource parsers. (Pitfall 3)
    3. Missing `X-Accel-Buffering: no` causes Nginx/Cloud Run to buffer SSE. (Pitfall 4)
    4. Zone data MUST NOT come from `LiveStore.getState().zones` тАФ it won't exist server-side. (Pitfall 5)
    5. The `?alt=sse` parameter must be in the endpoint URL. (Pitfall 1)
